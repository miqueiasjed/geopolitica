### Hook `useAiUso`:
```typescript
import { useQuery } from '@tanstack/react-query';
import { api } from '@/lib/api';
import { AiUsoData } from '@/types/ai';

export function useAiUso() {
  return useQuery<AiUsoData>({
    queryKey: ['admin', 'ai', 'uso'],
    queryFn: async () => {
      const response = await api.get<{ data: AiUsoData }>('/admin/ai/uso');
      return response.data.data;
    },
    staleTime: 60_000,   // 1 minuto — dados não mudam em tempo real
    refetchInterval: 120_000, // atualiza a cada 2 min enquanto a página está aberta
  });
}
```

### Regras de negócio aplicáveis:
- `staleTime: 60_000` evita refetch desnecessário.
- Sem `retry` extra — se o endpoint falhar, exibir estado de erro normal.

### Critérios de aceitação:
