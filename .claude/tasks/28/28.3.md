# Task 28.3 – Hook useProdutos.ts com helper resolveLinkCta

## Objetivo

## Especificação

### O que fazer:
```typescript
import { useQuery } from '@tanstack/react-query';
import type { Produto, StatusAddonUsuario } from '@/types/produto';
import { api } from '@/services/api';

const fetchMeusProdutos = (): Promise<Produto[]> =>
  api.get('/api/meus-produtos').then(r => r.data);

export function useProdutos() {
  return useQuery({
    queryKey: ['meus-produtos'],
    queryFn: fetchMeusProdutos,
    staleTime: 5 * 60 * 1000, // 5 minutos
  });
}

// Retorna o link correto para o CTA baseado no status atual do usuário
export function resolveLinkCta(produto: Produto): string | null {
  switch (produto.status_usuario) {
    case 'cancelado':
      return produto.link_reativar ?? produto.link_compra;
    case 'expirado':
      return produto.link_compra ?? produto.link_reativar;
    case 'reembolsado':
      return produto.link_compra;
    case null:
    default:
      return produto.link_compra;
  }
}

// Retorna o label correto para o CTA
export function resolveLabelCta(status: StatusAddonUsuario): string {
  switch (status) {
    case 'cancelado':  return 'Reativar';
    case 'expirado':   return 'Renovar';
    case 'reembolsado': return 'Assinar';
    case null:
    default:           return 'Adicionar';
  }
}
```

### Regras de negócio aplicáveis:
- `staleTime: 5min` evita requests desnecessários durante navegação
- `resolveLinkCta` para `cancelado` prefere `link_reativar`, com fallback para `link_compra`
- Retorna `null` quando ambos links são null (botão desabilitado)
