## Especificação

### O que fazer:
```typescript
// Status que o usuário tem para um produto específico
export type StatusAddonUsuario = 'ativo' | 'cancelado' | 'expirado' | 'reembolsado' | null;

// Produto retornado pelo endpoint /api/meus-produtos (visão do usuário)
export interface Produto {
  chave: string;
  nome: string;
  descricao: string | null;
  preco_label: string | null;
  link_compra: string | null;
  link_reativar: string | null;
  ativo: boolean;
  ordem: number;
  status_usuario: StatusAddonUsuario;
}

// Produto no contexto admin (CRUD) — sem status_usuario
export interface AdminProduto {
  id: number;
  chave: string;
  nome: string;
  descricao: string | null;
  preco_label: string | null;
  link_compra: string | null;
  link_reativar: string | null;
  ativo: boolean;
  ordem: number;
  created_at: string;
  updated_at: string;
}

// Payload para criar produto
export interface CriarProdutoPayload {
  chave: string;
  nome: string;
  descricao?: string | null;
  preco_label?: string | null;
  link_compra?: string | null;
  link_reativar?: string | null;
  ativo?: boolean;
  ordem?: number;
}

// Payload para atualizar produto (chave é imutável)
export type AtualizarProdutoPayload = Omit<CriarProdutoPayload, 'chave'>;
```

### Regras de negócio aplicáveis:
- `status_usuario: null` indica que o usuário nunca adquiriu o produto
- `chave` é imutável — `AtualizarProdutoPayload` não inclui esse campo
- `AdminProduto` inclui `id`, `created_at`, `updated_at` que `Produto` não tem

### Critérios de aceitação:
- [ ] Arquivo compila sem erros TypeScript
- [ ] `StatusAddonUsuario` aceita os 4 valores + null
- [ ] `AtualizarProdutoPayload` não inclui `chave`
- [ ] Tipos importáveis em outros arquivos: `import type { Produto } from '@/types/produto'`
