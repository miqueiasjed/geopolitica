## Especificação

### O que fazer:
**adminProdutos.ts:**
```typescript
import type { AdminProduto, CriarProdutoPayload, AtualizarProdutoPayload } from '@/types/produto';
import { api } from '@/services/api'; // ou padrão do projeto

export const adminProdutos = {
  listar: (): Promise<AdminProduto[]> =>
    api.get('/admin/produtos').then(r => r.data),

  criar: (payload: CriarProdutoPayload): Promise<AdminProduto> =>
    api.post('/admin/produtos', payload).then(r => r.data),

  atualizar: (id: number, payload: AtualizarProdutoPayload): Promise<AdminProduto> =>
    api.put(`/admin/produtos/${id}`, payload).then(r => r.data),

  excluir: (id: number): Promise<void> =>
    api.delete(`/admin/produtos/${id}`),

  importarAddons: (arquivo: File): Promise<{ importados: number; erros: Array<{ linha: number; motivo: string }> }> => {
    const form = new FormData();
    form.append('arquivo', arquivo);
    return api.post('/admin/assinantes/addons/importar', form).then(r => r.data);
  },

  exportarAddons: (): Promise<Blob> =>
    api.get('/admin/assinantes/addons/exportar', { responseType: 'blob' }).then(r => r.data),
};
```

**adminAssinanteAddons.ts:**
```typescript
// CRUD de addons por assinante individual
export const adminAssinanteAddons = {
  listar: (userId: number): Promise<AssinanteAddon[]> =>
    api.get(`/admin/assinantes/${userId}/addons`).then(r => r.data),

  adicionar: (userId: number, payload: AdicionarAddonPayload): Promise<AssinanteAddon> =>
    api.post(`/admin/assinantes/${userId}/addons`, payload).then(r => r.data),

  atualizar: (userId: number, addonId: number, payload: AtualizarAddonPayload): Promise<AssinanteAddon> =>
    api.put(`/admin/assinantes/${userId}/addons/${addonId}`, payload).then(r => r.data),

  remover: (userId: number, addonId: number): Promise<void> =>
    api.delete(`/admin/assinantes/${userId}/addons/${addonId}`),
};
```

Criar os tipos `AssinanteAddon`, `AdicionarAddonPayload` e `AtualizarAddonPayload` em `@/types/produto.ts` ou em um arquivo novo `@/types/assinanteAddon.ts`.

### Regras de negócio aplicáveis:

- `exportarAddons` deve retornar `Blob` para permitir download
